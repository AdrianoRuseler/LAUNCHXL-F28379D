837  // If HRPWM_PWMSYNC_SOURCE_PERIOD or HRPWM_PWMSYNC_SOURCE_ZERO
838  if(syncPulseSource < HRPWM_PWMSYNC_SOURCE_COMPC_UP)
839  {
840  HWREGH(base + HRPWM_O_HRPCTL) =
841  ((HWREGH(base + HRPWM_O_HRPCTL) &
842  ~(HRPWM_HRPCTL_PWMSYNCSELX_M | HRPWM_HRPCTL_PWMSYNCSEL)) |
843  ((uint16_t)syncPulseSource << 1U));
844  }
845  else
846  {
847  HWREGH(base + HRPWM_O_HRPCTL) =
848  ((HWREGH(base + HRPWM_O_HRPCTL) & ~HRPWM_HRPCTL_PWMSYNCSELX_M) |
849  ((uint16_t)syncPulseSource << HRPWM_HRPCTL_PWMSYNCSELX_S));
850  }
