108 static void Interrupt_defaultHandler(void)
109 {
110  uint16_t pieVect;
111  uint16_t vectID;
112 
113  //
114  // Calculate the vector ID. If the vector is in the lower PIE, it's the
115  // offset of the vector that was fetched (bits 7:1 of PIECTRL.PIEVECT)
116  // divided by two.
117  //
118  pieVect = HWREGH(PIECTRL_BASE + PIE_O_CTRL);
119  vectID = (pieVect & 0xFEU) >> 1U;
120 
121  //
122  // If the vector is in the upper PIE, the vector ID is 128 or higher.
123  //
124  if(pieVect >= 0x0E00U)
125  {
126  vectID += 128U;
127  }
128 
129  //
130  // Something has gone wrong. An interrupt without a proper registered
131  // handler function has occurred. To help you debug the issue, local
132  // variable vectID contains the vector ID of the interrupt that occurred.
133  //
134  ESTOP0;
135  for(;;)
136  {
137  ;
138  }
139 }

262 static inline void
263 Interrupt_register(uint32_t interruptNumber, void (*handler)(void))
264 {
265  uint32_t address;
266 
267  //
268  // Calculate appropriate address for the interrupt number
269  //
270  address = (uint32_t)PIEVECTTABLE_BASE +
271  (((interruptNumber & 0xFFFF0000U) >> 16U) * 2U);
272 
273  //
274  // Copy ISR address into PIE table
275  //
276  EALLOW;
277  HWREG(address) = (uint32_t)handler;
278  EDIS;
279 }
