356 static inline uint32_t
357 GPIO_readPin(uint32_t pin)
358 {
359  volatile uint32_t *gpioDataReg;
360 
361  //
362  // Check the arguments.
363  //
364  ASSERT(GPIO_isPinValid(pin));
365 
366  gpioDataReg = (uint32_t *)GPIODATA_BASE +
367  ((pin / 32U) * GPIO_DATA_REGS_STEP);
368 
369  return((gpioDataReg[GPIO_GPxDAT_INDEX] >> (pin % 32U)) & (uint32_t)0x1U);
370 }
