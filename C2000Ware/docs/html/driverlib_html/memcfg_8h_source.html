264 static inline void
265 MemCfg_setCLAMemType(uint32_t ramSections, MemCfg_CLAMemoryType claMemType)
266 {
267  //
268  // Check the arguments.
269  //
270  ASSERT((ramSections & MEMCFG_SECT_TYPE_MASK) == MEMCFG_SECT_TYPE_LS);
271 
272  //
273  // Write the CLA memory configuration to the appropriate register. Either
274  // set or clear the bit that determines the function of the RAM section as
275  // it relates to the CLA.
276  //
277  EALLOW;
278 
279  if(claMemType == MEMCFG_CLA_MEM_PROGRAM)
280  {
281  // Program memory
282  HWREG(MEMCFG_BASE + MEMCFG_O_LSXCLAPGM) |= ramSections;
283  }
284  else
285  {
286  // Data memory
287  HWREG(MEMCFG_BASE + MEMCFG_O_LSXCLAPGM) &= ~ramSections;
288  }
289 
290  EDIS;
291 }

401 MemCfg_getViolationInterruptStatus(void)
402 {
403  uint32_t status;
404

406  // Read and return RAM access status flags.
407  //
408  status = (HWREG(ACCESSPROTECTION_BASE + MEMCFG_O_NMAVFLG)) |

410  MEMCFG_MVIOL_SHIFT);
411 
412  return(status);
