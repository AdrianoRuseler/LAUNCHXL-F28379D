259 static inline DCSM_SecurityStatus
260 DCSM_getZone1CSMSecurityStatus(void)
261 {
262  uint16_t status;
263  DCSM_SecurityStatus returnStatus;
264  status = HWREGH(DCSM_Z1_BASE + DCSM_O_Z1_CR);
265 
266  //
267  // if ARMED bit is set and UNSECURED bit or ALLONE bit or both UNSECURED
268  // and ALLONE bits are set then CSM is unsecured. Else it is secure.
269  //
270  if(((status & DCSM_Z1_CR_ARMED) != 0U) &&
271  (((status & DCSM_Z1_CR_UNSECURE) != 0U) ||
272  ((status & DCSM_Z1_CR_ALLONE) != 0U )))
273  {
274  returnStatus = DCSM_STATUS_UNSECURE;
275  }
276  else if((status & DCSM_Z1_CR_ALLZERO) == DCSM_Z1_CR_ALLZERO)
277  {
278  returnStatus = DCSM_STATUS_LOCKED;
279  }
280  else
281  {
282  returnStatus = DCSM_STATUS_SECURE;
283  }
284 
285  return(returnStatus);
286 }
