394 static inline void CPUTimer_selectClockSource(uint32_t base,
395  CPUTimer_ClockSource source,
396  CPUTimer_Prescaler prescaler)
397 {
398  ASSERT(base == CPUTIMER2_BASE);
399 
400  //
401  // Set source and prescaler for CPU Timer 2
402  //
403  if(base == CPUTIMER2_BASE)
404  {
405  EALLOW;
406 
407  //
408  // Set Clock Source
409  //
410  HWREGH(CPUSYS_BASE + SYSCTL_O_TMR2CLKCTL) &=
411  ~SYSCTL_TMR2CLKCTL_TMR2CLKSRCSEL_M;
412 
413  HWREGH(CPUSYS_BASE + SYSCTL_O_TMR2CLKCTL) |= (uint16_t)source;
414 
415  //
416  // Set Clock Prescaler
417  //
418  HWREGH(CPUSYS_BASE + SYSCTL_O_TMR2CLKCTL) &=
419  ~SYSCTL_TMR2CLKCTL_TMR2CLKPRESCALE_M;
420 
421  HWREGH(CPUSYS_BASE + SYSCTL_O_TMR2CLKCTL) |= ((uint16_t)prescaler <<
422  SYSCTL_TMR2CLKCTL_TMR2CLKPRESCALE_S);
423 
424  EDIS;
425  }
